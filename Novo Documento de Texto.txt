  <script>
    // Dados de exemplo — aqui você pode puxar da sua API
    const products = [
      {id:1,name:'Dipirona 500mg - 10 comprimidos',price:5.50,category:'Analgésicos',img:'https://via.placeholder.com/300x200?text=Dipirona'},
      {id:2,name:'Paracetamol 500mg - 10 comprimidos',price:6.25,category:'Analgésicos',img:'https://via.placeholder.com/300x200?text=Paracetamol'},
      {id:3,name:'Vitamin C 1000mg - 60 cápsulas',price:29.90,category:'Vitaminas',img:'https://via.placeholder.com/300x200?text=Vitamina+C'},
      {id:4,name:'Xampu Anticaspa 200ml',price:18.00,category:'Higiene',img:'https://via.placeholder.com/300x200?text=Xampu'},
      {id:5,name:'Creme Dental 90g',price:8.50,category:'Higiene',img:'https://via.placeholder.com/300x200?text=Creme+Dental'},
      {id:6,name:'Máscara descartável 50un',price:45.00,category:'EPIs',img:'https://via.placeholder.com/300x200?text=Mascaras'}
    ];

    // Estado do carrinho
    const cart = {}; // {productId: qty}

    // Elementos
    const grid = document.querySelector('.grid');
    const categoryFilter = document.getElementById('categoryFilter');
    const searchInput = document.getElementById('searchInput');
    const searchBtn = document.getElementById('searchBtn');
    const cartCount = document.getElementById('cartCount');
    const viewCartBtn = document.getElementById('viewCartBtn');
    const cartFloating = document.getElementById('cartFloating');
    const cartItems = document.getElementById('cartItems');
    const cartTotal = document.getElementById('cartTotal');
    const closeCartBtn = document.getElementById('closeCartBtn');
    const checkoutBtn = document.getElementById('checkoutBtn');

    // Inicializa
    function init(){
      populateCategories();
      renderProducts(products);
      attachEvents();
      updateCartUI();
    }

    function populateCategories(){
      const cats = Array.from(new Set(products.map(p=>p.category)));
      cats.forEach(c=>{
        const opt = document.createElement('option'); opt.value=c; opt.textContent=c; categoryFilter.appendChild(opt);
      })
    }

    function renderProducts(list){
      grid.innerHTML='';
      if(list.length===0){grid.innerHTML='<div style="padding:20px;font-size:14px;color:var(--muted)">Nenhum produto encontrado.</div>';return}
      list.forEach(p=>{
        const card = document.createElement('article'); card.className='card';
        card.innerHTML = `
          <img src="${p.img}" alt="${p.name}" />
          <h3>${p.name}</h3>
          <div class="price">R$ ${p.price.toFixed(2)}</div>
          <div style="font-size:12px;color:var(--muted);margin-top:6px">${p.category}</div>
          <div class="actions">
            <button class="btn btn-outline" data-id="${p.id}">Detalhes</button>
            <button class="btn btn-primary" data-add="${p.id}">Adicionar</button>
          </div>
        `;
        grid.appendChild(card);
      })
    }

    function attachEvents(){
      searchBtn.addEventListener('click',()=>applyFilters());
      searchInput.addEventListener('keydown',(e)=>{ if(e.key==='Enter') applyFilters(); });
      categoryFilter.addEventListener('change',()=>applyFilters());

      grid.addEventListener('click',(e)=>{
        const id = e.target.getAttribute('data-add');
        if(id) addToCart(Number(id));
      })

      viewCartBtn.addEventListener('click',()=>{ cartFloating.style.display='block'; renderCart(); });
      closeCartBtn.addEventListener('click',()=>cartFloating.style.display='none');
      checkoutBtn.addEventListener('click',()=>{ alert('Simulação: redirecionar para página de pagamento / confirmação.'); });

      document.getElementById('sendMsg').addEventListener('click',()=>{
        alert('Mensagem enviada — aqui você integraria com seu e-mail ou API.');
      });
      document.getElementById('clearMsg').addEventListener('click',()=>{
        document.getElementById('name').value='';document.getElementById('email').value='';document.getElementById('message').value='';
      })
    }

    function applyFilters(){
      const q = searchInput.value.trim().toLowerCase();
      const cat = categoryFilter.value;
      const filtered = products.filter(p=>{
        const matchQ = q==='' || p.name.toLowerCase().includes(q) || p.category.toLowerCase().includes(q);
        const matchCat = cat==='' || p.category===cat;
        return matchQ && matchCat;
      })
      renderProducts(filtered);
    }

    function addToCart(id){
      cart[id] = (cart[id]||0) + 1;
      updateCartUI();
      // feedback rápido
      const prod = products.find(p=>p.id===id);
      const toast = document.createElement('div');
      toast.textContent = `${prod.name} adicionado ao carrinho`;
      toast.style.position='fixed';toast.style.left='50%';toast.style.transform='translateX(-50%)';toast.style.bottom='90px';toast.style.background='rgba(0,0,0,0.8)';toast.style.color='white';toast.style.padding='8px 12px';toast.style.borderRadius='8px';document.body.appendChild(toast);
      setTimeout(()=>toast.remove(),1400);
    }

    function updateCartUI(){
      const count = Object.values(cart).reduce((s,n)=>s+n,0);
      cartCount.textContent = count;
    }

    function renderCart(){
      cartItems.innerHTML='';
      let total = 0;
      if(Object.keys(cart).length===0){ cartItems.innerHTML='<div style="color:var(--muted)">Carrinho vazio.</div>'; cartTotal.textContent='R$ 0,00'; return }
      Object.entries(cart).forEach(([id,qty])=>{
        const p = products.find(x=>x.id===Number(id));
        const row = document.createElement('div'); row.className='cart-item';
        row.innerHTML = `
          <img src="${p.img}" alt="${p.name}" />
          <div style="flex:1">
            <div style="font-weight:700">${p.name}</div>
            <div style="font-size:13px;color:var(--muted)">R$ ${p.price.toFixed(2)} x ${qty}</div>
          </div>
          <div style="display:flex;flex-direction:column;gap:6px;align-items:flex-end">
            <button class="btn btn-outline" data-minus="${p.id}">-</button>
            <button class="btn btn-outline" data-plus="${p.id}">+</button>
            <button class="btn" data-remove="${p.id}" style="font-size:12px;color:#b00;background:transparent;border:0">Remover</button>
          </div>
        `;
        cartItems.appendChild(row);
        total += p.price * qty;
      })
      cartTotal.textContent = `R$ ${total.toFixed(2)}`;

      // eventos para + - remover
      cartItems.querySelectorAll('[data-plus]').forEach(btn=>btn.addEventListener('click',e=>{ const id=Number(e.target.getAttribute('data-plus')); cart[id]=(cart[id]||0)+1; renderCart(); updateCartUI(); }));
      cartItems.querySelectorAll('[data-minus]').forEach(btn=>btn.addEventListener('click',e=>{ const id=Number(e.target.getAttribute('data-minus')); if(!cart[id]) return; cart[id]--; if(cart[id]<=0) delete cart[id]; renderCart(); updateCartUI(); }));
      cartItems.querySelectorAll('[data-remove]').forEach(btn=>btn.addEventListener('click',e=>{ const id=Number(e.target.getAttribute('data-remove')); delete cart[id]; renderCart(); updateCartUI(); }));
    }

    // inicialização
    init();
  </script>
